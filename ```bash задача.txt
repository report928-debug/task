```bash
sudo apt update
sudo apt install -y docker.io docker-compose
sudo systemctl enable --now docker
```

---
```bash
mkdir -p pg_cluster/master pg_cluster/replica1 pg_cluster/replica2
cd pg_cluster
```

---
```yaml
version: '3.9'

services:
  master:
    image: postgres:16
    container_name: pg-master
    environment:
      POSTGRES_PASSWORD: masterpass
      POSTGRES_USER: master
      POSTGRES_DB: appdb
    volumes:
      - ./master:/var/lib/postgresql/data
    ports:
      - "8765:5432"
    networks:
      - pgnet

  replica1:
    image: postgres:16
    container_name: pg-replica1
    environment:
      POSTGRES_PASSWORD: replpass
      POSTGRES_USER: repl
      POSTGRES_DB: appdb
      POSTGRES_PRIMARY_HOST: master
      POSTGRES_PRIMARY_PORT: 5432
      POSTGRES_PRIMARY_USER: master
      POSTGRES_PRIMARY_PASSWORD: masterpass
      POSTGRES_REPLICATION_MODE: replica
    depends_on:
      - master
    volumes:
      - ./replica1:/var/lib/postgresql/data
    networks:
      - pgnet

  replica2:
    image: postgres:16
    container_name: pg-replica2
    environment:
      POSTGRES_PASSWORD: replpass
      POSTGRES_USER: repl
      POSTGRES_DB: appdb
      POSTGRES_PRIMARY_HOST: master
      POSTGRES_PRIMARY_PORT: 5432
      POSTGRES_PRIMARY_USER: master
      POSTGRES_PRIMARY_PASSWORD: masterpass
      POSTGRES_REPLICATION_MODE: replica
    depends_on:
      - master
    volumes:
      - ./replica2:/var/lib/postgresql/data
    networks:
      - pgnet

networks:
  pgnet:
```

---
```conf
listen_addresses = '*'
wal_level = replica
max_wal_senders = 10
wal_keep_size = 64
```

Создай `master/pg_hba.conf`:

```conf
host replication all 0.0.0.0/0 md5
host all all 0.0.0.0/0 md5
```

---
```bash
docker-compose up -d
```

Проверяем:

```bash
docker ps
```

---
```bash
psql -h localhost -p 8765 -U master -d appdb
```

Создаём таблицу:

```sql
CREATE TABLE test (id serial PRIMARY KEY, txt text);
INSERT INTO test (txt) VALUES ('hello');
```
```bash
docker exec -it pg-replica1 psql -U repl -d appdb -c "SELECT * FROM test;"
```
